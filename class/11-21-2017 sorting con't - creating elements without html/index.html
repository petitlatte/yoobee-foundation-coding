JS indexOf() method

										
											// Checking if something exists in an array without indexOf()
											var arr = ['Fish','Tigers','Sheep', 'Tigers', 'Horses'],
											found = false;

											for(var i= 0, l = arr.length; i< l; i++){
												if(arr[i] === 'Tigers'){
													found = true;
													console.log('run!');
												}
											}

											// By utilzing -1 in the indexOf syntax we can scan the array

JS filter() method (repeated from previous slides but a refresher):

										
											//Filtering data without the filter method
											var arr = [
											    {"name":"Bk's", "count": 2},
											    {"name":"Macca's", "count": 5},
											    {"name":"Pizza Hutt", "count": 3},
											    {"name":"Fruit and Veges", "count": 16}
											];

											//Notice the empty array push pattern. Very common and
											//useful for pushing and filtering a new set of data. 
											var newArr = [];

											for(var i= 0, l = arr.length; i< l; i++){
											    if(arr[i].name === "Fruit and Veges" ){
													newArr.push(arr[i]);
												}
											}

											console.log("Filter results:",newArr);

											//Here we use the filter method
											var arr = [
											    {"name":"rabbits", "count": 11},
											    {"name":"cats", "count": 2},
											    {"name":"dogs", "count": 5},
											    {"name":"rabbits", "count": 35},
											    {"name":"parrots", "count": 3},
											    {"name":"rabbits", "count": 2},
											    {"name":"rabbits", "count": 4}
											];
											    
											var newArr = arr.filter(function(item){
											    return item.name === "rabbits";
											});
											
											// so the variable newArr contains an updated filtered array of data
											console.log("Filter results:",newArr);

JS forEach() method

The forEach method can be more readible, succinct than using a normal for loop. It also provides functional scope so variables are private when used withn a forEach method. (Variables are not scoped/private when used within a for loop.)

										
											var arr = [1,2,3,4,5,6,7,8];

											// Uses the usual "for" loop to iterate
											for(var i= 0, l = arr.length; i< l; i++){
												console.log(arr[i]);
											}

											

											//Uses forEach to iterate
											arr.forEach(function(item,index){
												console.log(item);
											});

jQuery $.each()

A jQuery method that has the ability to loop through array items and objects.

										
											// Looping and printing from arrays and an object with jQuery's $.each

												//First the array
												var arr = [22,10,15,121,15,44,48,47,55,91,12,10];
												
												$.each(arr, function( index, value ) {
												  console.log( index + ": " + value );
												});

												//We can also use $.each to loop over an object
												var objA = {
												  "flammable": "inflammable",
												  "duh": "no duh",
												  "really": "yes",
												  "still reading?": "thats a bummer man!"
												};

												$.each( objA, function( key, value ) {
												  console.log( key + ": " + value );
												});
